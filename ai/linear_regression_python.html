X = df[["YearsExperience"]]
y = df["Salary"]
model = LinearRegression().fit(X, y)
y_pred = model.predict(X)
print("Slope: %.4f" % model.coef_[0])
print("Intercept: %.4f" % model.intercept_)
print("R-squared: %.4f" % r2_score(y, y_pred))
